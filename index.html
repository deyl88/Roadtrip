function makeBoardSet() { const used = new Set(); const pools = getPools(); if (gameMode === "kid") { const out = []; for (let i = 0; i < 4; i++) { const item = pickFromPool(pools.kid, used); out.push(item); used.add(item.label); } return shuffle(out); } if (gameMode === "hard") { const out = []; out.push(pickFromPool(pools.easy, used)); used.add(out[out.length - 1].label); out.push(pickFromPool(pools.medium, used)); used.add(out[out.length - 1].label); out.push(pickFromPool(pools.hard, used)); used.add(out[out.length - 1].label); out.push(pickFromPool(pools.hard, used)); used.add(out[out.length - 1].label); return shuffle(out); } // CLASSIC MODE const selected = []; // Always 2 easy selected.push(pickFromPool(pools.easy, used)); used.add(selected[selected.length - 1].label); selected.push(pickFromPool(pools.easy, used)); used.add(selected[selected.length - 1].label); // Always 1 medium selected.push(pickFromPool(pools.medium, used)); used.add(selected[selected.length - 1].label); // Last slot: usually easy/medium, occasionally hard const roll = Math.random(); if (roll < 0.25) { selected.push(pickFromPool(pools.hard, used)); } else if (roll < 0.60) { selected.push(pickFromPool(pools.medium, used)); } else { selected.push(pickFromPool(pools.easy, used)); } used.add(selected[selected.length - 1].label); return shuffle(selected); }
